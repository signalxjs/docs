```tsx
import { component, computed } from 'sigx';

const PriceCalculator = component(({ signal }) => {
    const state = signal({ price: 100, quantity: 2 });
    
    // Computed is a signal - access via .value
    const total = computed(() =>
        state.price * state.quantity
    );
    
    return () => (
        <div>
            <p>Total: ${total.value}</p>
        </div>
    );
});
```
