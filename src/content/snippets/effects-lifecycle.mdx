```tsx
import { component, effect, onMounted, onUnmounted } from 'sigx';

const DataFetcher = component(({ signal, props }) => {
    const state = signal({ data: null as any });
    
    onMounted(() => console.log('Component mounted'));
    onUnmounted(() => console.log('Cleanup'));
    
    // Effect runs when userId changes
    effect(() => {
        fetch(`/api/users/${props.userId}`)
            .then(r => r.json())
            .then(d => state.data = d);
    });
    
    return () => <div>{JSON.stringify(state.data)}</div>;
});
```
