# Range

Range slider component for selecting numeric values with model binding support.

## Import

```tsx
import { Range, Row, Col, Heading, Text } from '@sigx/daisyui';
```

## Basic Usage

Use `model` for two-way binding:

```tsx live preview
import { component, render } from 'sigx';
import { Range, Card, Text } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const form = signal({ value: 50 });
    
    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <Range 
                    model={() => form.value}
                    color="primary"
                />
                
                <Text size="sm" class="opacity-70">
                    Value: {form.value}
                </Text>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Price Range Filter

```tsx live preview
import { component, render } from 'sigx';
import { Range, Card, Badge, Text, Row, Col, Heading } from '@sigx/daisyui';
const Demo = component(({ signal }) => {
    const filter = signal({ minPrice: 0, maxPrice: 500 });
    
    return () => (
        <Card>
            <Card.Body class="space-y-6">
                <Heading level={3} size="lg">Price Filter</Heading>
                
                <Col gap="4">
                    <Col gap="2">
                        <Row justify="between" class="mb-2">
                            <Text as="span" size="sm">Min Price</Text>
                            <Badge variant="ghost">${filter.minPrice}</Badge>
                        </Row>
                        <Range 
                            model={() => filter.minPrice}
                            min={0}
                            max={filter.maxPrice}
                            step={10}
                            color="primary"
                        />
                    </Col>
                    
                    <Col gap="2">
                        <Row justify="between" class="mb-2">
                            <Text as="span" size="sm">Max Price</Text>
                            <Badge variant="ghost">${filter.maxPrice}</Badge>
                        </Row>
                        <Range 
                            model={() => filter.maxPrice}
                            min={filter.minPrice}
                            max={1000}
                            step={10}
                            color="primary"
                        />
                    </Col>
                </Col>
                
                <Text align="center" weight="medium">
                    ${filter.minPrice} - ${filter.maxPrice}
                </Text>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `model` | `() => signal.property` | - | Model binding for numeric value |
| `name` | `string` | - | Range name for form submission |
| `min` | `number` | `0` | Minimum value |
| `max` | `number` | `100` | Maximum value |
| `step` | `number` | `1` | Step increment |
| `size` | `'xs' \| 'sm' \| 'md' \| 'lg'` | `'md'` | Range slider size |
| `color` | `'primary' \| 'secondary' \| 'accent' \| 'info' \| 'success' \| 'warning' \| 'error'` | - | Slider color |
| `disabled` | `boolean` | `false` | Disable the range slider |
| `class` | `string` | - | Additional CSS classes |

## Events

| Event | Type | Description |
|-------|------|-------------|
| `onChange` | `(value: number) => void` | Fired when value changes |
