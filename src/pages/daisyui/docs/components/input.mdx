# Input

Text input component for single-line text entry with model binding support.

## Import

```tsx
import { Input, Col } from '@sigx/daisyui';
```

## Basic Usage

Use `model` for two-way binding:

```tsx live preview
import { component, render } from 'sigx';
import { Input, Card, Col, Text } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const form = signal({ 
        name: '',
        email: ''
    });
    
    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <Input 
                    model={() => form.name}
                    placeholder="Enter your name"
                    variant="bordered"
                />
                <Input 
                    model={() => form.email}
                    type="email"
                    placeholder="Enter your email"
                    variant="bordered"
                />
                <Col gap="1">
                    <Text size="sm" class="opacity-70">Name: {form.name || '(empty)'}</Text>
                    <Text size="sm" class="opacity-70">Email: {form.email || '(empty)'}</Text>
                </Col>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## With FormField

Wrap inputs with `FormField` for labels, helper text, and error messages:

```tsx live preview
import { component, render } from 'sigx';
import { Input, FormField, Card } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const form = signal({ email: '', password: '' });
    const emailError = () => form.email && !form.email.includes('@') ? 'Please enter a valid email' : '';
    
    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <FormField 
                    label="Email" 
                    required 
                    error={emailError()}
                >
                    <Input 
                        model={() => form.email}
                        type="email"
                        placeholder="you@example.com"
                        color={emailError() ? 'error' : undefined}
                        variant="bordered"
                    />
                </FormField>
                
                <FormField 
                    label="Password" 
                    hint="Must be at least 8 characters"
                >
                    <Input 
                        model={() => form.password}
                        type="password" 
                        placeholder="••••••••" 
                        variant="bordered"
                    />
                </FormField>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Number Input

For number inputs, the model value is automatically converted to a number:

```tsx live preview

    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <FormField label="Quantity">

                        model={() => form.quantity}

                        model={() => form.price}
                        type="number" 
                        variant="bordered"
                    />
                </FormField>
```

## Login Form Example

```tsx live preview
import { component, render } from 'sigx';
import { Input, FormField, Button, Card, Checkbox, Divider, Heading } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const form = signal({ 
        email: '', 
        password: '',
        remember: false
    });
    
    const errors = signal({
        email: '',
        password: ''
    });
    
    const validate = () => {
        let valid = true;
        
        if (!form.email) {
            errors.email = 'Email is required';
            valid = false;
        } else if (!form.email.includes('@')) {
            errors.email = 'Invalid email format';
            valid = false;
        } else {
            errors.email = '';
        }
        
        if (!form.password) {
            errors.password = 'Password is required';
            valid = false;
        } else if (form.password.length < 8) {
            errors.password = 'Password must be at least 8 characters';
            valid = false;
        } else {
            errors.password = '';
        }
        
        return valid;
    };
    
    const handleSubmit = () => {
        if (validate()) {
            alert('Login successful!');
        }
    };
    
    return () => (
        <Card class="max-w-md mx-auto">
            <Card.Body class="space-y-4">
                <Heading level={2} size="xl" align="center">Sign In</Heading>
                
                <FormField label="Email" required error={errors.email}>
                    <Input 
                        model={() => form.email}
                        type="email"
                        placeholder="you@example.com"
                        color={errors.email ? 'error' : undefined}
                        variant="bordered"
                    />
                </FormField>
                
                <FormField label="Password" required error={errors.password}>
                    <Input 
                        model={() => form.password}
                        type="password"
                        placeholder="••••••••"
                        color={errors.password ? 'error' : undefined}
                        variant="bordered"
                    />
                </FormField>
                
                <Checkbox 
                    model={() => form.remember}
                    label="Remember me"
                    color="primary"
                />
                
                <Button variant="primary" block onClick={handleSubmit}>
                    Sign In
                </Button>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Autocomplete Control

Use `autoComplete` to control browser autofill behavior:

```tsx live preview
import { component, render } from 'sigx';
import { Input, FormField, Card } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const form = signal({ email: '', password: '' });
    
    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <FormField label="Email">
                    <Input 
                        model={() => form.email}
                        type="email"
                        placeholder="user@example.com"
                        autoComplete="off"
                        variant="bordered"
                    />
                </FormField>
                
                <FormField label="New Password">
                    <Input 
                        model={() => form.password}
                        type="password"
                        placeholder="(unchanged)"
                        autoComplete="new-password"
                        variant="bordered"
                    />
                </FormField>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

Common `autoComplete` values: `'off'`, `'on'`, `'email'`, `'username'`, `'current-password'`, `'new-password'`, `'name'`, `'tel'`, `'address-line1'`.

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `model` | `() => signal.property` | - | Model binding for two-way sync |
| `name` | `string` | - | Input name for form submission |
| `type` | `'text' \| 'email' \| 'password' \| 'number' \| 'search' \| 'url' \| 'tel'` | `'text'` | Input type |
| `size` | `'xs' \| 'sm' \| 'md' \| 'lg'` | `'md'` | Input size |
| `variant` | `'bordered' \| 'ghost'` | - | Visual variant |
| `color` | `'primary' \| 'secondary' \| 'accent' \| 'info' \| 'success' \| 'warning' \| 'error'` | - | Border color |
| `placeholder` | `string` | - | Placeholder text |
| `disabled` | `boolean` | `false` | Disable the input |
| `readOnly` | `boolean` | `false` | Read-only state |
| `maxLength` | `number` | - | Maximum character length |
| `autoComplete` | `string` | - | Browser autocomplete hint (`'off'`, `'new-password'`, `'email'`, etc.) |
| `class` | `string` | - | Additional CSS classes |

## Events

| Event | Type | Description |
|-------|------|-------------|
| `onInput` | `(e: InputEvent) => void` | Fired on each keystroke (raw DOM event) |
| `onChange` | `(e: Event) => void` | Fired when focus leaves the input |
| `onFocus` | `(e: FocusEvent) => void` | Fired when input gains focus |
| `onBlur` | `(e: FocusEvent) => void` | Fired when input loses focus |
