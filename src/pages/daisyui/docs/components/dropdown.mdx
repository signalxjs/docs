# Dropdown

Dropdown component for creating action menus, context menus, and select-like interfaces.

## Import

```tsx
import { Dropdown } from '@sigx/daisyui';
```

## Basic Dropdown

```tsx live preview
import { component, render } from 'sigx';
import { Dropdown, Button, Menu, Card, Row, Text } from '@sigx/daisyui';

const Demo = component(() => {
    return () => (
        <Card>
            <Card.Body class="py-12">
                <Row justify="center">
                    <Dropdown
                        slots={{
                            trigger: () => <Button>Click me</Button>
                        }}
                    >
                        <Menu.Item><Text as="span">Item 1</Text></Menu.Item>
                        <Menu.Item><Text as="span">Item 2</Text></Menu.Item>
                        <Menu.Item><Text as="span">Item 3</Text></Menu.Item>
                    </Dropdown>
                </Row>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Dropdown Positions

```tsx live preview
import { component, render } from 'sigx';
import { Dropdown, Button, Menu, Card, Row, Text } from '@sigx/daisyui';

const Demo = component(() => {
    const positions = ['bottom', 'top', 'left', 'right'] as const;
    
    return () => (
        <Card>
            <Card.Body class="py-12">
                <Row wrap justify="center" gap="4">
                    {positions.map(pos => (
                        <Dropdown 
                            position={pos}
                            menuClass="w-40"
                            slots={{
                                trigger: () => <Button>{pos}</Button>
                            }}
                        >
                            <Menu.Item><Text as="span">Option 1</Text></Menu.Item>
                            <Menu.Item><Text as="span">Option 2</Text></Menu.Item>
                        </Dropdown>
                    ))}
                </Row>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Context Menu Example

```tsx live preview
import { component, render } from 'sigx';
import { Dropdown, Menu, Card, Badge, Col, Row, Text } from '@sigx/daisyui';

const Demo = component(() => {
    const items = [
        { id: 1, name: 'Document 1.pdf', type: 'PDF' },
        { id: 2, name: 'Image.png', type: 'Image' },
        { id: 3, name: 'Report.xlsx', type: 'Excel' }
    ];
    
    return () => (
        <Card>
            <Card.Body>
                <Col gap="2">
                    <Text size="sm" color="muted" margin={{ bottom: "4" }}>Hover on items to see menu:</Text>
                    
                    {items.map(item => (
                        <Dropdown 
                            hover
                            menuClass="w-48"
                            slots={{
                                trigger: () => (
                                    <Row align="center" gap="3" class="p-3 bg-base-200 rounded-lg cursor-pointer hover:bg-base-300 w-full">
                                        <Text as="span" size="2xl">📄</Text>
                                        <Col gap="1" class="flex-1">
                                            <Text as="span" weight="medium">{item.name}</Text>
                                            <Badge variant="ghost" size="sm">{item.type}</Badge>
                                        </Col>
                                    </Row>
                                )
                            }}
                        >
                            <Menu.Item><Text as="span">Open</Text></Menu.Item>
                            <Menu.Item><Text as="span">Download</Text></Menu.Item>
                            <Menu.Item><Text as="span">Rename</Text></Menu.Item>
                            <Menu.Item><Text as="span">Share</Text></Menu.Item>
                            <Col class="border-t border-base-300 mt-1 pt-1">
                                <Menu.Item><Text as="span" color="error">Delete</Text></Menu.Item>
                            </Col>
                        </Dropdown>
                    ))}
                </Col>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `position` | `'end' \| 'top' \| 'bottom' \| 'left' \| 'right'` | - | Dropdown position |
| `hover` | `boolean` | `false` | Open on hover instead of click |
| `menuClass` | `string` | - | Additional CSS classes for the dropdown menu |
| `class` | `string` | - | Additional CSS classes for the container |

## Slots

| Slot | Description |
|------|-------------|
| `trigger` | The element that opens the dropdown (can be any component) |
| `default` | Dropdown menu content (typically Menu.Item components) |
