# Select

Select component for dropdown selection with model binding support.

## Import

```tsx
import { Select, Col } from '@sigx/daisyui';
```

## Basic Usage

Use `model` for two-way binding:

```tsx live preview
import { component, render } from 'sigx';
import { Select, Card, Text } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const form = signal({ country: '' });
    
    const countries = [
        { value: 'us', label: 'United States' },
        { value: 'uk', label: 'United Kingdom' },
        { value: 'ca', label: 'Canada' },
        { value: 'au', label: 'Australia' },
        { value: 'de', label: 'Germany' }
    ];
    
    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <Select 
                    model={() => form.country}
                    options={countries}
                    placeholder="Select a country"
                    variant="bordered"
                />
                
                <Text size="sm" class="opacity-70">
                    Selected: {form.country || '(none)'}
                </Text>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## With FormField

Wrap with `FormField` for labels and validation:

```tsx live preview
import { component, render } from 'sigx';
import { Select, FormField, Card } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const form = signal({ category: '' });
    const hasError = () => !form.category;
    
    const categories = [
        { value: 'tech', label: 'Technology' },
        { value: 'design', label: 'Design' },
        { value: 'business', label: 'Business' },
        { value: 'other', label: 'Other' }
    ];
    
    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <FormField 
                    label="Category" 
                    required 
                    error={hasError() ? 'Please select a category' : ''}
                >
                    <Select 
                        model={() => form.category}
                        options={categories}
                        placeholder="Choose a category"
                        color={hasError() ? 'error' : undefined}
                        variant="bordered"
                    />
                </FormField>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `model` | `() => signal.property` | - | Model binding for two-way sync |
| `name` | `string` | - | Select name for form submission |
| `options` | `Array<{ value, label, disabled? }>` | - | Select options |
| `placeholder` | `string` | - | Placeholder text (shown as disabled first option) |
| `size` | `'xs' \| 'sm' \| 'md' \| 'lg'` | `'md'` | Select size |
| `variant` | `'bordered' \| 'ghost'` | - | Visual variant |
| `color` | `'primary' \| 'secondary' \| 'accent' \| 'info' \| 'success' \| 'warning' \| 'error'` | - | Border color |
| `disabled` | `boolean` | `false` | Disable the select |
| `class` | `string` | - | Additional CSS classes |

## Events

| Event | Type | Description |
|-------|------|-------------|
| `onChange` | `(value: string) => void` | Fired when selection changes |
