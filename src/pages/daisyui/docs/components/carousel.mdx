# Carousel

Carousel for displaying sliding content with navigation.

## Import

```tsx
import { Carousel, CarouselItem } from '@sigx/daisyui';
```

## Basic Carousel

```tsx live preview
import { component, render } from 'sigx';
import { Carousel, CarouselItem, Card, Text } from '@sigx/daisyui';

const Demo = component(() => {
    const images = [
        'https://picsum.photos/400/200?random=1',
        'https://picsum.photos/400/200?random=2',
        'https://picsum.photos/400/200?random=3',
        'https://picsum.photos/400/200?random=4'
    ];
    
    return () => (
        <Card>
            <Carousel class="rounded-box">
                {images.map((src, i) => (
                    <CarouselItem id={`slide${i}`}>
                        <img src={src} class="w-full" alt={`Slide ${i + 1}`} />
                    </CarouselItem>
                ))}
            </Carousel>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Carousel with Navigation

```tsx live preview
import { component, render } from 'sigx';
import { Carousel, CarouselItem, Card, Button, Row, Heading } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const state = signal({ currentSlide: 0 });
    const slides = [1, 2, 3, 4, 5];
    
    const goTo = (index) => {
        state.currentSlide = index;
        document.getElementById(`item${index}`).scrollIntoView({ behavior: 'smooth', inline: 'start' });
    };
    
    const next = () => goTo((state.currentSlide + 1) % slides.length);
    const prev = () => goTo((state.currentSlide - 1 + slides.length) % slides.length);
    
    return () => (
        <Card class="p-4 space-y-4">
            <Carousel class="rounded-box">
                {slides.map((num, i) => (
                    <CarouselItem id={`item${i}`} class="w-full">
                        <Card class="w-full h-48 bg-base-200">
                            <Card.Body class="items-center justify-center">
                                <Heading level={3} size="3xl">Slide {num}</Heading>
                            </Card.Body>
                        </Card>
                    </CarouselItem>
                ))}
            </Carousel>
            
                <Row justify="between" align="center">
                    <Button variant="circle" onClick={prev}>❮</Button>
                    <Row gap="2">
                        {slides.map((_, i) => (
                            <Button
                                size="xs"
                                variant={state.currentSlide === i ? 'primary' : 'ghost'}
                                onClick={() => goTo(i)}
                            >
                                {i + 1}
                            </Button>
                        ))}
                    </Row>
                    <Button variant="circle" onClick={next}>❯</Button>
            </Row>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Auto-play Carousel

```tsx live preview
import { component, render, onMount, onUnmount } from 'sigx';
import { Carousel, CarouselItem, Card, Button, Badge, Row, Text } from '@sigx/daisyui';

const Demo = component(({ signal }) => {
    const state = signal({ 
        currentSlide: 0,
        playing: true
    });
    
    const slides = [
        { text: 'Slide 1', bg: 'from-primary to-secondary' },
        { text: 'Slide 2', bg: 'from-secondary to-accent' },
        { text: 'Slide 3', bg: 'from-accent to-info' },
        { text: 'Slide 4', bg: 'from-info to-success' },
        { text: 'Slide 5', bg: 'from-success to-warning' }
    ];
    
    let intervalId;
    
    const startAutoplay = () => {
        intervalId = setInterval(() => {
            if (state.playing) {
                state.currentSlide = (state.currentSlide + 1) % slides.length;
            }
        }, 3000);
    };
    
    const stopAutoplay = () => {
        if (intervalId) clearInterval(intervalId);
    };
    
    onMount(() => startAutoplay());
    onUnmount(() => stopAutoplay());
    
    const togglePlay = () => {
        state.playing = !state.playing;
    };
    
    return () => (
        <Card>
            <Card.Body class="space-y-4">
                <div class="relative w-full h-48 overflow-hidden rounded-box">
                    {slides.map((slide, i) => (
                        <div 
                            class={`absolute inset-0 transition-opacity duration-500 bg-gradient-to-r ${slide.bg} flex items-center justify-center text-white text-3xl font-bold ${
                                state.currentSlide === i ? 'opacity-100' : 'opacity-0'
                            }`}
                        >
                            <Text size="3xl" weight="bold" color="white">{slide.text}</Text>
                        </div>
                    ))}
                </div>
                
                <Row justify="between" align="center">
                    <Button 
                        variant="ghost" 
                        size="sm"
                        onClick={togglePlay}
                    >
                        {state.playing ? '⏸ Pause' : '▶ Play'}
                    </Button>
                    
                    <Row gap="2">
                        {slides.map((_, i) => (
                            <Button
                                size="xs"
                                variant={state.currentSlide === i ? 'primary' : 'ghost'}
                                onClick={() => state.currentSlide = i}
                            >
                                {i + 1}
                            </Button>
                        ))}
                    </Row>
                    
                    <Badge variant="ghost">
                        {state.currentSlide + 1} / {slides.length}
                    </Badge>
                </Row>
            </Card.Body>
        </Card>
    );
});

render(<Demo />, "#sandbox");
```

## Props

### Carousel Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `vertical` | `boolean` | `false` | Vertical scrolling |
| `snap` | `'start' \| 'center' \| 'end'` | `'start'` | Snap alignment |
| `variant` | `'full'` | - | Full-bleed mode |
| `class` | `string` | - | Additional CSS classes |

### Carousel Slots

| Slot | Description |
|------|-------------|
| `default` | CarouselItem components |

### CarouselItem Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `id` | `string` | - | Unique identifier for navigation |
| `class` | `string` | - | Additional CSS classes |
